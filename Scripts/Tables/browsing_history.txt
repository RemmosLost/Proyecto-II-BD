CREATE TABLE `per`.`browsing_history` (
  `id_browsing_history` INT NOT NULL AUTO_INCREMENT,
  `date` DATE NOT NULL,
  `creation_date` DATE NULL,
  `creation_user` VARCHAR(45) NULL,
  `last_update_date` DATE NULL,
  `last_update_user` VARCHAR(45) NULL,
  PRIMARY KEY (`id_browsing_history`),
  UNIQUE INDEX `id_browsing_history_UNIQUE` (`id_browsing_history` ASC) VISIBLE)
COMMENT = 'Table that stores browsing_history information.';
DROP TRIGGER IF EXISTS `per`.`browsing_history_BEFORE_UPDATE`;

DELIMITER $$
USE `per`$$
CREATE DEFINER = CURRENT_USER TRIGGER `per`.`browsing_history_BEFORE_UPDATE` BEFORE UPDATE ON `browsing_history` FOR EACH ROW
BEGIN
IF NEW.last_update_date IS NULL OR NEW.last_update_user IS NULL THEN
        SET NEW.last_update_date = CURDATE();
        SET NEW.last_update_user = SYSTEM_USER();
END IF;
END$$
DELIMITER ;
DROP TRIGGER IF EXISTS `per`.`browsing_history_BEFORE_INSERT`;

DELIMITER $$
USE `per`$$
CREATE DEFINER = CURRENT_USER TRIGGER `per`.`browsing_history_BEFORE_INSERT` BEFORE INSERT ON `browsing_history` FOR EACH ROW
BEGIN
IF NEW.creation_date IS NULL OR NEW.creation_user IS NULL THEN 
        SET NEW.creation_date = CURDATE();
        SET NEW.creation_user = SYSTEM_USER();
    END IF;
END$$
DELIMITER ;
